mysql> create procedure clapercentage()
    -> begin
    -> declare s1,s2,s3,tot int(3);
    -> declare per float(5,2);
    -> declare grade_st varchar(20);
    -> declare i,cnt,id int(10);
    -> SET i = 0;
    -> select count(*) into cnt from student;
    -> while i < cnt
    -> do
    -> select sno,sub1,sub2,sub3 into id,s1,s2,s3 from student limit i,1;
    -> set tot = (s1+s2+s3);
    -> set per = tot/3;
    -> if per > 40 then
    -> set grade_st = 'PASS';
    -> else
    -> set grade_st = 'FAIL';
    -> end if;
    -> select tot,per,grade_st;
    -> update student set total= tot,percentage=per,grade=grade_st where sno = id;
    -> set i =i+1;
    -> end while;
    -> end//
